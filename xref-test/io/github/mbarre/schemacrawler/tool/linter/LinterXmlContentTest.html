44  public class LinterXmlContentTest extends BaseLintTest {
45  
46  	private static final String CHANGE_LOG_XML_CHECK = "src/test/db/liquibase/xmlContentCheck/db.changelog.xml";
47  	private static PostgreSqlDatabase database;
48  
49  	@BeforeClass
50  	public static void  init(){
51  		database = new PostgreSqlDatabase();
52  		database.setUp(CHANGE_LOG_XML_CHECK);
53  	}
54  
55  	@Test
56  	public void testLint() throws Exception{
57  
58  		final LinterRegistry registry = new LinterRegistry();
59  		Assert.assertTrue(registry.hasLinter(LinterXmlContent.class.getName()));
60  
61  		final SchemaCrawlerOptions options = new SchemaCrawlerOptions();
62  		options.setSchemaInfoLevel(SchemaInfoLevelBuilder.standard());
63  		options.setTableNamePattern("test_xml");
64  
65  		Connection connection = DriverManager.getConnection(PostgreSqlDatabase.CONNECTION_STRING,
66  				PostgreSqlDatabase.USER_NAME, database.getPostgresPassword());
67  
68  		List<LintWrapper> lints = executeToJsonAndConvertToLintList(options, connection);
69  		Assert.assertEquals(1,lints.size());
70  		Assert.assertEquals(LinterXmlContent.class.getName(), lints.get(0).getId());
71  		Assert.assertEquals("public.test_xml.content", lints.get(0).getValue());
72  		Assert.assertEquals("should be XML type.", lints.get(0).getDescription());
73  		Assert.assertEquals("high", lints.get(0).getSeverity());
74  
75  	}
