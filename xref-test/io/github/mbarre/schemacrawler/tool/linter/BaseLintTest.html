52      protected List<LintWrapper> executeToJsonAndConvertToLintList(SchemaCrawlerOptions options, Connection connection) throws Exception {
53  
54          final Executable executable = new SchemaCrawlerExecutable("lint");
55  
56          final Path linterConfigsFile = FileSystems.getDefault().getPath("", this.getClass().getClassLoader().getResource("schemacrawler-linter-configs-test.xml").getPath());
57          final LintOptionsBuilder optionsBuilder = new LintOptionsBuilder();
58          optionsBuilder.withLinterConfigs(linterConfigsFile.toString());
59          executable.setAdditionalConfiguration(optionsBuilder.toConfig());
60  
61          Path out = Paths.get("target/test_"+this.getClass().getSimpleName()+".json");
62          OutputOptions outputOptions = new OutputOptions(TextOutputFormat.json, out);
63          outputOptions.setOutputFile(Paths.get("target/test_"+this.getClass().getSimpleName()+".json"));
64  
65          executable.setOutputOptions(outputOptions);
66          executable.setSchemaCrawlerOptions(options);
67          executable.execute(connection);
68          
69          File output = new File(out.toString());
70          String data = IOUtils.toString(new FileInputStream(output));
71          Assert.assertNotNull(data);
72          JSONObject json = new JSONObject(data.toString().substring(1, data.toString().length() - 2));
73         
74          List<LintWrapper>lints = new ArrayList<>();
75          
76          if( json.get("table_lints") instanceof  JSONObject) {
77  
78              Assert.assertNotNull(json.getJSONObject("table_lints"));
79              JSONArray jsonLints = json.getJSONObject("table_lints").getJSONArray("lints");
80              Assert.assertNotNull(jsonLints);
81  
82              if(options.getTableNamePattern() != null  && ! options.getTableNamePattern().isEmpty())
83                  Assert.assertEquals(options.getTableNamePattern(), json.getJSONObject("table_lints").getString("name"));
84  
85              for (int i = 0; i < jsonLints.length(); i++) {
86                  if(!"databasechangelog".equals(json.getJSONObject("table_lints").getString("name")) &&
87                          !"databasechangeloglock".equals(json.getJSONObject("table_lints").getString("name")))
88                      lints.add(createLintWrapper(json.getJSONObject("table_lints").getString("name"), jsonLints.getJSONObject(i)));
89              }
90          }
91          else{
92              Assert.assertNotNull(json.getJSONArray("table_lints"));
93              JSONArray jsonTableLints = json.getJSONArray("table_lints");
94  
95              for (int i = 0; i < jsonTableLints.length(); i++) {
96                  JSONArray jsonLints = jsonTableLints.getJSONObject(i).getJSONArray("lints");
97                  Assert.assertNotNull(jsonLints);
98  
99                  if(options.getTableNamePattern() != null  && ! options.getTableNamePattern().isEmpty())
100                     Assert.assertEquals(options.getTableNamePattern(), json.getJSONObject("table_lints").getString("name"));
101 
102                 for (int j = 0; j < jsonLints.length(); j++) {
103                     if(!"databasechangelog".equals(jsonTableLints.getJSONObject(i).getString("name")) &&
104                             !"databasechangeloglock".equals(jsonTableLints.getJSONObject(i).getString("name")))
105                         lints.add(createLintWrapper(jsonTableLints.getJSONObject(i).getString("name"), jsonLints.getJSONObject(j)));
106                 }
107             }
108         }
109         return lints;
110     }

112     private LintWrapper createLintWrapper(String tableName, JSONObject jsonLint){
113 
114         LintWrapper lint = new LintWrapper();
115         lint.setId(jsonLint.getString("id"));
116         Assert.assertNotNull(lint.getId());
117         lint.setValue(jsonLint.getString("value").trim());
118         Assert.assertNotNull(lint.getValue());
119         lint.setDescription(jsonLint.getString("description").trim());
120         Assert.assertNotNull(lint.getDescription());
121         lint.setSeverity(jsonLint.getString("severity").trim());
122         Assert.assertNotNull(lint.getSeverity());
123         lint.setTableName(tableName);
124 
125         return lint;
126     }
