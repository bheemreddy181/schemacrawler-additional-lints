43  	@Test
44  	public void testLint() throws Exception{
45  
46  		final LinterRegistry registry = new LinterRegistry();
47  		Assert.assertTrue(registry.hasLinter(LinterTableWithNoPrimaryKey.class.getName()));
48  
49  		final SchemaCrawlerOptions options = new SchemaCrawlerOptions();
50  		// Set what details are required in the schema - this affects the
51  		// time taken to crawl the schema
52  		options.setSchemaInfoLevel(SchemaInfoLevelBuilder.standard());
53  		options.setTableNamePattern("test_primary_key");
54  		
55  		Connection connection = DriverManager.getConnection(PostgreSqlDatabase.CONNECTION_STRING, 
56  				PostgreSqlDatabase.USER_NAME, database.getPostgresPassword());
57  		
58  		final Executable executable = new SchemaCrawlerExecutable("lint");
59  		try (StringBuilderWriter out = new StringBuilderWriter()) {
60  			OutputOptions outputOptions = new OutputOptions(TextOutputFormat.json,out);
61  			executable.setOutputOptions(outputOptions);
62  			executable.setSchemaCrawlerOptions(options);
63  			executable.execute(connection);
64  
65  			Assert.assertNotNull(out.toString());
66  			JSONObject json = new JSONObject(out.toString().substring(1, out.toString().length()-1)) ;
67  			Assert.assertNotNull(json.getJSONObject("table_lints"));
68  			Assert.assertEquals("test_primary_key", json.getJSONObject("table_lints").getString("name"));
69  
70  			JSONArray lints = json.getJSONObject("table_lints").getJSONArray("lints");
71  
72  			boolean lintDectected = false;
73  			
74  			for (int i=0; i < lints.length(); i++) {
75  				if(LinterTableWithNoPrimaryKey.class.getName().equals(lints.getJSONObject(i).getString("id"))){
76  					if("test_primary_key".equals(lints.getJSONObject(i).getString("value").trim())){
77  						Assert.assertEquals("table should have a primary key", lints.getJSONObject(i).getString("description").trim());
78  						Assert.assertEquals("high", lints.getJSONObject(i).getString("severity").trim());
79  						lintDectected = true;
80  					}
81  					else{
82  						Assert.fail("Not expected error detected :"+lints.getJSONObject(i).getString("value").trim());
83  					}
84  				}
85  			}
86  
87  			Assert.assertTrue("Some expected errors have not been detected.", lintDectected);
88  		}
