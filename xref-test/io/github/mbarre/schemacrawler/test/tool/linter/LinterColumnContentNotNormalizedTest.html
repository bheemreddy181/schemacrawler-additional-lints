80              Assert.assertNotNull(json.getJSONObject("table_lints"));

82              JSONArray lints = json.getJSONObject("table_lints").getJSONArray("lints");
83                          // now we have the json array, let's filter only the one we want in our lint
84              // lint id : [io.github.mbarre.schemacrawler.tool.linter.LinterColumnContentNotNormalized]
85              boolean lint1Dectected = false;
86              for (int i = 0; i < lints.length(); i++) {
87                  // be sure we are on the right lint
88                  if (LinterColumnContentNotNormalized.class.getName().equals(lints.getJSONObject(i).getString("id"))) {
89                      if ("public.test_normalized.content".equals(lints.getJSONObject(i).getString("value").trim())) {

91                          Assert.assertEquals("high", lints.getJSONObject(i).getString("severity").trim());
