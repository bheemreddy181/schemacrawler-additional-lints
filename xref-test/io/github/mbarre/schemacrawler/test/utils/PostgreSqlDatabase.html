65      public void setProperties(){
66          this.properties = new Properties();
67  	InputStream input;
68          LOG.info("Looking for <test/resources/test.properties> file, with key <postgres.password> ...");
69          try{
70              
71              input = getClass().getClassLoader().getResourceAsStream("test.properties");
72              if(input == null){
73                  LOG.info("Not able to open <test/resources/test.properties> file,"
74                          + " skipping and setting to default password");
75                  setPostgresPassword(DEFAULT_PASSWORD);
76                  return;
77              }
78  
79              
80              properties.load(input);
81              LOG.info("Found <test/resources/test.properties> file");
82              LOG.info("Found postgres password : <" + properties.getProperty("postgres.password") +">");
83              if(properties.getProperty("postgres.password") != null){
84                  LOG.info("<postgres.password> found...");
85                  setPostgresPassword(properties.getProperty("postgres.password"));
86              }
87              else{
88                  LOG.info("<postgres.password> nof found : default empty value will be used.");
89                  setPostgresPassword(DEFAULT_PASSWORD);
90              }
91          }
92          catch(FileNotFoundException ex){
93              LOG.error("Could not find <test.properties> conf file... will use default", ex);
94              this.properties = null;
95              setPostgresPassword(DEFAULT_PASSWORD);
96          }
97          catch(IOException ex){
98              LOG.error("Could not find <test.properties> conf file... will use default", ex);
99              this.properties = null;
100             setPostgresPassword(DEFAULT_PASSWORD);
101         }
102     }

110     private void createTables(String changelog) {
111 
112 	Connection holdingConnection;
113 	try {
114 	    ResourceAccessor resourceAccessor = new FileSystemResourceAccessor();
115 
116 	    holdingConnection = getConnectionImpl(USER_NAME, getPostgresPassword());
117 	    JdbcConnection conn = new JdbcConnection(holdingConnection);
118 
119 	    PostgresDatabase database = new PostgresDatabase();
120 	    database.setDefaultSchemaName("public");
121 	    database.setConnection(conn);
122 
123 	    liquibase = new Liquibase(changelog, resourceAccessor, database);
124 	    liquibase.dropAll();
125 	    liquibase.update("test");
126 
127 	    conn.close();
128 
129 	} catch (SQLException | LiquibaseException ex) {
130 	    LOG.error("Error during createTable step", ex);
131 	    throw new RuntimeException("Error during createTable step", ex);
132 	}
