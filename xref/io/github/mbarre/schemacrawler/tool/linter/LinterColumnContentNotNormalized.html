41  /**
42   * Test column contents normalization. Detect wether foreign table should have
43   * be used.
44   * @author salad74
45   */
46  public class LinterColumnContentNotNormalized extends BaseLinter {
47      
48      private static final Logger LOGGER = Logger.getLogger(LinterColumnContentNotNormalized.class.getName());
49      private static final String NB_REPEAT_TOLERANCE_CONFIG = "nbRepeatTolerance";
50      private static final String MIN_TEXT_COLUMN_SIZE_CONFIG = "minTextColumnSize";
51      
52      /**
53       * Repeat tolerance : 1 is the most agressive, duplicates occur since there
54       * are at least tow reps of the same value.
55       */
56      public static final int NB_REPEAT_TOLERANCE = 2;
57      
58      /**
59       * The minimal length of the text based column. We consider that a 1-length
60       * char based column is acceptable.
61       */
62      public static final int MIN_TEXT_COLUMN_SIZE = 2;
63      
64      private int nbRepeatTolerance;
65      private int minTextColumnSize;
66      
67      
68      /**
69       * Build the lint
70       */
71      public LinterColumnContentNotNormalized() {
72          setSeverity(LintSeverity.high);
73          nbRepeatTolerance = NB_REPEAT_TOLERANCE;
74          minTextColumnSize = MIN_TEXT_COLUMN_SIZE;
75      }
76      
77      /**
78       * Get the lint descrption
79       * @return The Lint description
80       */
81      @Override
82      public String getDescription() {
83          return getSummary();
84      }
85      
86      /**
87       * Get the Summary
88       * @return The lint summary
89       */
90      @Override
91      public String getSummary() {
92          return " should not have so many duplicates.";
93      }
94  
95      @Override
96      public void configure(Config config) {
97          nbRepeatTolerance = config.getIntegerValue(NB_REPEAT_TOLERANCE_CONFIG, NB_REPEAT_TOLERANCE);
98          minTextColumnSize = config.getIntegerValue(MIN_TEXT_COLUMN_SIZE_CONFIG, MIN_TEXT_COLUMN_SIZE);
99      }
100         
101     /**
102      * Tells wether a column is text based or not... and if minimal length
103      * requirements are met to make normalization computations.
104      *
105      * @param javaSqlType javaSqlType
106      * @param colSize colSize
107      * @param minColumnSize minColSize
108      * @return mustColumnBeTested
109      */
110     public static final boolean mustColumnBeTested(int javaSqlType, int colSize, int minColumnSize) {
111         if (LintUtils.isSqlTypeTextBased(javaSqlType)) {
112             // test minimal size to pass test
113             if(colSize > minColumnSize){
114                 LOGGER.log(Level.INFO, "Column min size requirement are met : <{0}> is greater than <{1}>", new Object[]{colSize, minColumnSize});
115                 return true;
116             }
117             else{
118                 return false;
119             }
120         } else {
121             return false;
122         }
123     }
124     
125     /**
126      * The lint that does the job
127      * @param table table
128      * @param connection the db connection
129      * @throws SchemaCrawlerException SchemaCrawlerException
130      */
131     @Override
132     protected void lint(final Table table, final Connection connection)
133             throws SchemaCrawlerException {
134         
135         LOGGER.log(Level.CONFIG, "<nbRepeatTolerance> parameter set to {0}",nbRepeatTolerance);
136         LOGGER.log(Level.CONFIG, "<minTextColumnSize> parameter set to {0}",minTextColumnSize);
137         
138         try (Statement stmt = connection.createStatement()){
139             String sql;
140             String tableName = table.getName().replaceAll("\"", "");
141             List<Column> columns = table.getColumns();
142             
143             for (Column column : columns) {
144                 if (LinterColumnContentNotNormalized.mustColumnBeTested(column.getColumnDataType().getJavaSqlType().getJavaSqlType(), column.getSize(), minTextColumnSize)) {
145                     // test based column, perform test
146                     LOGGER.log(Level.INFO, "Checking {0}...", column.getFullName());
147                     String columnName = column.getName().replaceAll("\"", "");
148                     sql = "select \"" + columnName + "\", count(*)  as counter from \"" + tableName + "\" where \"" + columnName + "\" is not null group by \"" + columnName + "\" having count(*) > " + nbRepeatTolerance + " order by count(*) desc";
149                     LOGGER.log(Level.CONFIG, "SQL : {0}", sql);
150                     
151                     try(ResultSet rs = stmt.executeQuery(sql)){
152                         while (rs.next()) {
153                             int nbRepeats = rs.getInt("counter");
154                             LOGGER.log(Level.CONFIG, "Found <{0}> repetitions of the same value <{1}> in <{2}>", new Object[]{nbRepeats, rs.getString(1), column});
155                             if(nbRepeats > nbRepeatTolerance){
156                                 LOGGER.log(Level.CONFIG, "Adding lint as nbRepeats exceeds tolerance ({0} > {1} )", new Object[]{nbRepeats, nbRepeatTolerance});
157                                 addLint(table, "Found <" + nbRepeats + "> repetitions of the same value <" + rs.getString(1) + "> in <" + column + ">", column.getFullName());
158                             }
159                         }
160                     } catch (SQLException ex) {
161                         LOGGER.severe(ex.getMessage());
162                         throw new SchemaCrawlerException(ex.getMessage(), ex);
163                     }
164                 } else {
165                     // no text based column, skip test
166                     LOGGER.log(Level.CONFIG, "<{0}> is not text based : normalize test will be skipped.", column);
167                 }
168                 
169             }
170         } catch (SQLException ex) {
171             LOGGER.severe(ex.getMessage());
172             throw new SchemaCrawlerException(ex.getMessage(), ex);
173         }
174         
175     }
