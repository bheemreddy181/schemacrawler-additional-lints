25  import java.sql.Connection;
26  import java.sql.ResultSet;
27  import java.sql.SQLException;
28  import java.sql.Statement;
29  import java.util.List;
30  import java.util.Set;
31  import java.util.logging.Level;
32  import java.util.logging.Logger;
33  
34  import io.github.mbarre.schemacrawler.utils.LintUtils;
35  import schemacrawler.schema.Column;
36  import schemacrawler.schema.Table;
37  import schemacrawler.schemacrawler.Config;
38  import schemacrawler.schemacrawler.SchemaCrawlerException;
39  import schemacrawler.tools.lint.BaseLinter;
40  import schemacrawler.tools.lint.LintSeverity;
41  
42  public class LinterLeftSpacePadding extends BaseLinter {
43  	public static final String SAMPLE_SIZE_PARAM = "sampleSize";
44  	private static final Logger LOGGER = Logger.getLogger(LinterLeftSpacePadding.class.getName());
45  	private Integer sampleSize;
46  
47  	/**
48  	 * The lint that parses and test Json content
49  	 */
50  	public LinterLeftSpacePadding() {
51  		sampleSize = 1000;
52  		setSeverity(LintSeverity.high);
53  	}
54  
55  	@Override
56  	public void configure(Config config) {
57  		sampleSize = config.getIntegerValue(SAMPLE_SIZE_PARAM, 1000);
58  	}
59  
60  	@Override
61  	public String getSummary() {
62  		return "Column with left space padding";
63  	}
64  
65  	@Override
66  	protected void lint(Table table, Connection connection) throws SchemaCrawlerException {
67  		try (Statement stmt = connection.createStatement()){
68  
69  			LOGGER.log(Level.CONFIG, "<" + SAMPLE_SIZE_PARAM + "> parameter set to {0}%", sampleSize);
70  
71  			String sql;
72  			String columnName;
73  			String valueFound = "";
74  			String tableName = table.getName().replaceAll("\"", "");
75  
76  			List<Column> columns = getColumns(table);
77  
78  			Long totalRows = 0L;
79  			totalRows = LintUtils.getTableSize(stmt, tableName);
80  			LOGGER.log(Level.INFO, "totalrows=" + totalRows);
81  			Set<Long> sampleIndexes = LintUtils.generateSample(sampleSize, totalRows);
82  			LOGGER.log(Level.INFO, "sampleIndexes=" + sampleIndexes);
83  
84  			for (Column column : columns) {
85  				if(LintUtils.isSqlTypeTextBased(column.getColumnDataType().getJavaSqlType().getVendorTypeNumber())){
86  
87  					columnName = column.getName().replaceAll("\"", "");
88  					sql = "select distinct \"" + columnName + "\" from \"" + tableName + "\"" ;
89  
90  					LOGGER.log(Level.CONFIG, "SQL : {0}", sql);
91  
92  					try (ResultSet rs = stmt.executeQuery(sql)) {
93  						Long i = 0L;
94  						while (rs.next()) {
95  							if (sampleIndexes.contains(i)) {
96  								if (rs.getObject(columnName) != null) {
97  									valueFound = rs.getObject(columnName).toString();
98  								}
99  
100 								if(valueFound.length() > 0 && Character.isWhitespace(valueFound.charAt(0))){
101 									addLint(table, getDescription(), column.getFullName());
102 								}
103 							}
104 							i++;
105 						}
106 					}catch (SQLException ex) {
107 						LOGGER.severe(ex.getMessage());
108 						throw new SchemaCrawlerException(ex.getMessage(), ex);
109 					}
110 
111 				}
112 			}
113 
114 		}catch (SQLException ex) {
115 			LOGGER.log(Level.SEVERE, ex.getMessage());
116 			throw new SchemaCrawlerException(ex.getMessage(), ex);
117 		}
118 	}
119 }
