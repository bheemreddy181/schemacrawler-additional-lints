25  import java.sql.Connection;
26  import java.util.logging.Level;
27  import java.util.logging.Logger;
28  
29  import schemacrawler.schema.Column;
30  import schemacrawler.schema.ForeignKey;
31  import schemacrawler.schema.ForeignKeyColumnReference;
32  import schemacrawler.schema.Table;
33  import schemacrawler.schema.View;
34  import schemacrawler.schemacrawler.Config;
35  import schemacrawler.tools.lint.BaseLinter;
36  import schemacrawler.tools.lint.LintSeverity;
37  
38  import static java.util.Objects.requireNonNull;
39  
40  /**
41   *  Check that foreign key name has right prefix or suffix name
42   * Created by barmi83 on 12/09/16.
43   */
44  
45  public class LinterForeignKeyName  extends BaseLinter {
46  
47      private static final Logger LOGGER = Logger.getLogger(LinterForeignKeyName.class.getName());
48  
49      public static final String PREFIX_CONFIG_PARAM = "foreignKeyPrefix";
50      public static final String SUFFIX_CONFIG_PARAM = "foreignKeySuffix";
51  
52      private String prefix;
53      private String suffix;
54  
55      public LinterForeignKeyName() {
56          prefix = "";
57          suffix = "";
58          setSeverity(LintSeverity.medium);
59      }
60  
61      @Override
62      public String getSummary() {
63          return "FK name does not contain required prefix or suffix.";
64      }
65  
66      /**
67       * Get the description
68       * @return the description
69       */
70      @Override
71      public String getDescription() {
72          return "Foreign key data type does not match Primary key";
73      }
74  
75      @Override
76      public void configure(Config config) {
77          prefix = config.getStringValue(PREFIX_CONFIG_PARAM, "");
78          suffix = config.getStringValue(SUFFIX_CONFIG_PARAM, "");
79      }
80  
81      @Override
82      protected void lint(Table table, Connection connection) {
83  
84          LOGGER.log(Level.CONFIG, "<"+PREFIX_CONFIG_PARAM+"> parameter set to {0}%", prefix);
85          LOGGER.log(Level.CONFIG, "<"+SUFFIX_CONFIG_PARAM+"> parameter set to {0}%", suffix);
86  
87          requireNonNull(table, "No table provided");
88          findMismatchedForeignKeys(table);
89  
90      }
91  
92      private void findMismatchedForeignKeys(final Table table){
93  
94          if(prefix.isEmpty() && suffix.isEmpty())
95              return;
96  
97          if (table != null && !(table instanceof View))
98          {
99              for (final ForeignKey foreignKey: table.getImportedForeignKeys())
100             {
101                 for (final ForeignKeyColumnReference columnReference: foreignKey)
102                 {
103                     final Column fkColumn = columnReference.getForeignKeyColumn();
104                     if (fkColumn != null && !fkColumn.getName().startsWith(prefix))
105                     {
106                         addTableLint(table, "Foreign key does not start with prefix \"" + prefix +"\"", foreignKey);
107 
108                     }
109                     if (fkColumn != null && !fkColumn.getName().endsWith(suffix))
110                     {
111                         addTableLint(table, "Foreign key does not ends with suffix \"" + suffix +"\"", foreignKey);
112                     }
113                 }
114             }
115         }
116     }
